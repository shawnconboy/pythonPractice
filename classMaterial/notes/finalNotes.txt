ðŸ”¹ 2. Helper Functions
------------------------
def clearScreen():
    os.system('cls' if os.name == 'nt' else 'clear')

def header():
    print("Welcome to [Business Name]!")
    print("-" * 45)

def printMenu():
    print("Menu")
    print("-" * 45)
    for item in [inventory_dict]:
        print(f"{item:<30}${[inventory_dict][item]:>10.2f}")

def transition():
    clearScreen()
    header()
    printMenu()

ðŸ”¹ 3. Main Program Loop
--------------------------
response = "y"
while response == "y":
    clearScreen()
    total = 0
    transition()
    
    name = input("Please enter your name: ")
    
    cartItems = []
    cartPrices = []

    addItem = "y"
    while addItem == "y":
        transition()
        choice = input("Please make a selection: ").title()
        
        if choice in [inventory_dict]:
            price = [inventory_dict][choice]
            cartItems.append(choice)
            cartPrices.append(price)
            total += price
            print(f"{choice} added! Current total is ${total:.2f}")
        else:
            print("Invalid selection.")

        addItem = input("Add another item? (Y/N): ").lower()
